echarts图表语法

记录一些echarts图表语法

写在开头

详细了解请戳我 👉 echarts官网

  1. 获取 ECharts
  • 在 ECharts 的 GitHub 获取。
  • 通过 npm 获取 echarts，npm install echarts --save，详见“[在 webpack 中使用 echarts](https://echarts.apache.org/zh/tutorial.html#在 webpack 中使用 ECharts)”
  • 通过 jsDelivr 等 CDN 引入
  1. 引入 ECharts
<script src="echarts.min.js"></script>
  1. 绘制一个简单的图表
<body>
    <div id="main" style="width: 600px;height:400px;"></div>
</body>
<script>
    var myChart = echarts.init(document.getElementById('main'));
    var option = {
        xAxis: {
            data: ["衬衫", "羊毛衫", "雪纺衫", "裤子", "高跟鞋", "袜子"]
        },
        yAxis: {},
        series: [{
            type: 'bar',
            data: [5, 20, 36, 10, 10, 20]
        }]
    };
    myChart.setOption(option);
</script>

饼图模板

<body>
    <div id="main" style="width: 600px;height:400px;"></div>
    <script>
        var myChart = echarts.init(document.getElementById('main'));


        var option = { //指定图表的配置项和数据
            backgroundColor: '', // 背景颜色
            tooltip: { //鼠标移入提示
                trigger: 'item', //圆形图专用
                formatter: '{a} <br/>{b} : {c} ({d}%)' //显示百分比
            },
            color: [], //区块颜色
            visualMap: { // 左下角略缩图
                show: true, //显示与否
                inRange: {} //在选中范围中的视觉元素
            },
            series: [{ //图表数据
                name: '', //鼠标移入区块显示的字
                type: 'pie', //属性：圆
                radius: '80%', //调整圆大小
                center: ['50%', '50%'], //圆心位置
                startAngle: 90, //起始角度
                clockwise = true, //是否顺时针排布
                roseType: 'radius', //'radius'扇区，'area'纯圆
                animationType: 'scale', //初始动画效果，可选
                animationEasing: 'quinticIn', //初始动画的缓动效果。可选
                data: [{}], //数据
                /*
                数据可以点函数
                data: [{}].sort(//逆时针从大到小
                    function(a, b) {
                    return a.value - b.value;}
                    ),
                */
                animationDelay: function() {
                    /*
                    初始动画的延迟，支持回调函数，
                    可以通过返回每个不同的数据
                    来实现戏剧性的动画效果啊 
                    */
                    return Math.random() * 500;
                },
            }]
        };
        myChart.setOption(option); //使用刚指定的配置项和数据显示图表。
    </script>
</body>

饼图实例

<body>
<div id="main" style="width: 600px;height:400px;"></div>
<script>
    var myChart = echarts.init(document.getElementById('main'));
    //指定图表的配置项和数据
    var option = {
        // 背景颜色
        backgroundColor: '#fff',
        //鼠标移入提示
        tooltip: {
            trigger: 'item', //圆形图专用
            formatter: '{a} <br/>{b} : {c} ({d}%)' //显示百分比
        },
        //区块颜色
        color: ['#fff', '#006c00', '#ed8884', '#ff9f7f', '#0096ff', '#9fe6b8', '#ff61ff', 'red'],
        // 左下角略缩图
        visualMap: {
            show: true,
            inRange: {
                // colorLightness: [0, 1],
                colorSaturation: [0, 1]
            }
        },
        //图表数据
        series: [{
            name: '访问来源',
            type: 'pie',
            //调整圆大小
            radius: '80%',
            //圆心位置
            center: ['50%', '50%'],
            data: [ { value: 180, name: '湖北' },
                    { value: 88, name: '云南' },
                    { value: 120, name: '北京' },
                    { value: 130, name: '山东' },
                    { value: 135, name: '河北' },
                    { value: 66, name: '江苏' },
                    { value: 130, name: '浙江' },
                    { value: 150, name: '四川' }].
            sort(function(a, b) {
                 //逆时针从大到小
                return a.value - b.value;
            }),
            roseType: 'radius', //'radius'扇区，'area'纯原
            animationType: 'scale', //初始动画效果，可选
            animationEasing: 'quinticIn', //初始动画的缓动效果。可选
            animationDelay: function(idx) {
                //初始动画的延迟，支持回调函数，
                //可以通过每个数据返回不同的 delay 时间实现更戏剧的初始动画效果。    
                return Math.random() * 500;
            }
        }]
    };
    //使用刚指定的配置项和数据显示图表。
    myChart.setOption(option);
</script>
